	- IF PROCESS TYPE IS "DEQUEUE"
		CLIENT : DEQUEUE
		SERVER : queue_name
		SERVER : data
		SERVER : END
	
	- IF PROCESS TYPE IS "DEQUEUE COMPLETED"
		CLIENT : DEQUEUE_COMPLETED
		CLIENT : queue_name
		SERVER : END

	- IF PROCESS TYPE IS "CLOSE"
		CLIENT : CLOSE
